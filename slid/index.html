Soft Delete

Registros nunca são removidos do banco — recebem um deleted_at

Como funciona

  • DELETE seta deleted_at com timestamp
  • GET lista filtra WHERE deleted_at IS NULL
  • Deletar registro já deletado retorna 404
  • Dado permanece no banco para auditoria

Entidades afetadas

  • User
  • Lead
  • Stage
Coluna deleted_at TIMESTAMP NULL adicionada via migration

Controller

Recebe request, delega ao Service, retorna ApiResponse

class LeadController
{
    public function __construct(
        private LeadService $leadService
    ) {}

    public function list(): JsonResponse
    {
        $leads = $this->leadService->findAll();
        if (empty($leads)) {
            return ApiResponse::notFound('Nenhum lead encontrado.');
        }
        return ApiResponse::success($leads);
    }

    public function delete(int $id): JsonResponse
    {
        try {
            $this->leadService->delete($id);
            return ApiResponse::deleted();        // 204 No Content
        } catch (EntityNotFoundException $e) {
            return ApiResponse::notFound($e->getMessage());
        }
    }
}

Service + Validator

Validação fluente, lógica de negócio e soft delete

class LeadService
{
    public function create(array $data): array
    {
        (new Validator($data))
            ->requiredInt('userId', 'O userId é obrigatório.')
            ->required('name', 'O nome é obrigatório.')
            ->email('email', 'Um e-mail válido é obrigatório.')
            ->positiveNumber('value', 'O valor deve ser positivo.')
            ->throw();

        $lead = new Lead();
        $lead->setUser($user)->setStage($stage);
        $lead->setName($data['name'])->setValue($data['value']);

        $this->leadRepo->save($lead);
        return $this->toArray($lead);
    }

    public function delete(int $id): void
    {
        $lead = $this->leadRepo->findById($id);
        if (null === $lead || null !== $lead->getDeletedAt()) {
            throw new EntityNotFoundException("Lead {$id} não encontrado.");
        }
        $lead->setDeletedAt(new \DateTimeImmutable());
        $this->leadRepo->save($lead);
    }
}

Repository

Acesso ao banco via Doctrine ORM com filtro de soft delete

class LeadRepository extends ServiceEntityRepository
{
    public function __construct(ManagerRegistry $registry)
    {
        parent::__construct($registry, Lead::class);
    }

    public function findById(int $id): ?Lead
    {
        return $this->find($id);
    }

    public function findAllActive(): array
    {
        return $this->createQueryBuilder('l')
            ->where('l.deletedAt IS NULL')
            ->getQuery()
            ->getResult();
    }

    public function save(Lead $lead): void
    {
        $this->getEntityManager()->persist($lead);
        $this->getEntityManager()->flush();
    }
}

// Unitário: verifica que deletedAt é setado e save() é chamado
public function testDeleteSetsDeletedAtAndCallsSave(): void
{
    $lead = $this->existingLead();
    $leadRepo = $this->createMock(LeadRepository::class);
    $leadRepo->method('findById')->willReturn($lead);
    $leadRepo->expects($this->once())->method('save');

    (new LeadService($leadRepo, $this->userRepo, $this->stageRepo))
        ->delete(1);

    $this->assertNotNull($lead->getDeletedAt());
}

// Integração: verifica que registro deletado sai da listagem
public function testDeletedLeadIsExcludedFromList(): void
