Classe ApiResponse — respostas JSON consistentes

class ApiResponse
{
    public static function success(array $data, int $status = 200): JsonResponse {
        return new JsonResponse(['data' => $data], $status);
    }

    public static function created(array $data): JsonResponse {
        return new JsonResponse(['data' => $data], 201);
    }

    public static function notFound(string $message): JsonResponse {
        return new JsonResponse(['message' => $message], 404);
    }

    public static function deleted(): JsonResponse {
        return new JsonResponse(null, 204);
    }

    public static function validationError(array $errors): JsonResponse {
        return new JsonResponse(
            ['message' => 'Dados invalidos.', 'errors' => $errors], 422);
    }
}

Controller

Delegacao total ao Service + tratamento de excecoes

final class LeadController extends AbstractController
{
    public function __construct(private readonly LeadService $leadService) {}

    public function create(Request $request): JsonResponse {
        $data = '' !== $request->getContent() ? $request->toArray() : [];
        try {
            return ApiResponse::created($this->leadService->create($data));
        } catch (ValidationException $e) {
            return ApiResponse::validationError($e->getErrors());
        }
    }

    public function update(int $id, Request $request): JsonResponse {
        $data = '' !== $request->getContent() ? $request->toArray() : [];
        try {
            return ApiResponse::success($this->leadService->update($id, $data));
        } catch (EntityNotFoundException $e) {
            return ApiResponse::notFound($e->getMessage());
        } catch (ValidationException $e) {
            return ApiResponse::validationError($e->getErrors());
        }
    }

    public function delete(int $id): JsonResponse {
        try {
            $this->leadService->delete($id);
            return ApiResponse::deleted();
        } catch (EntityNotFoundException $e) {
            return ApiResponse::notFound($e->getMessage());
        }
    }
}

Service — Create

Validacao completa + persistencia

public function create(array $data): array
{
    $user  = isset($data['userId']) && is_int($data['userId'])
        ? $this->userRepo->findById($data['userId']) : null;
    $stage = isset($data['stageId']) && is_int($data['stageId'])
        ? $this->stageRepo->findById($data['stageId']) : null;

    (new Validator($data))
        ->requiredInt('userId', 'O userId e obrigatorio.')
        ->requiredInt('stageId', 'O stageId e obrigatorio.')
        ->required('name', 'O nome e obrigatorio.')
        ->required('company', 'A empresa e obrigatoria.')
        ->email('email', 'Um e-mail valido e obrigatorio.')
        ->required('phone', 'O telefone e obrigatorio.')
        ->positiveNumber('value', 'O valor deve ser positivo.')
        ->check('userId', null !== $user, 'Usuario nao encontrado.')
        ->check('stageId', null !== $stage, 'Estagio nao encontrado.')
        ->throw();

    $lead = new Lead();
    $lead->setUser($user);
    $lead->setStage($stage);
    $lead->setName($data['name']);
    // ... demais campos
    $this->leadRepo->save($lead);

    return $this->toArray($lead);
}

Service — Soft Delete

public function delete(int $id): void
{
    $lead = $this->leadRepo->findById($id);

    if (null === $lead || null !== $lead->getDeletedAt()) {
        throw new EntityNotFoundException(
            "Lead {$id} nao encontrado."
        );
    }

    $lead->setDeletedAt(new \DateTimeImmutable());
    $this->leadRepo->save($lead);
}
Soft delete via campo deleted_at: o registro nao e removido fisicamente do banco, apenas marcado com a data de exclusao. O findAllActive() filtra registros deletados.

Validator — Interface Fluente

Encadeamento de metodos com acumulo de erros

class Validator {
    private array $errors = [];
    public function __construct(private readonly array $data) {}

    public function required(string $field, string $message): static {
        if (empty($this->data[$field])) $this->errors[$field] = $message;
        return $this;
    }
    public function email(string $field, string $message): static {
        if (!$this->hasError($field) && $this->has($field)
            && !filter_var($this->data[$field], FILTER_VALIDATE_EMAIL))
            $this->errors[$field] = $message;
        return $this;
    }
    public function check(string $field, bool $cond, string $msg): static {
        if (!$this->hasError($field) && $this->has($field) && !$cond)
            $this->errors[$field] = $msg;
        return $this;
    }
    public function throw(): void {
        if (!empty($this->errors)) throw new ValidationException($this->errors);
    }
}

class LeadServiceTest extends TestCase {
    private LeadService $service;

    protected function setUp(): void {
        $this->leadRepo  = $this->createStub(LeadRepository::class);
        $this->userRepo  = $this->createStub(UserRepository::class);
        $this->stageRepo = $this->createStub(StageRepository::class);
        $this->service   = new LeadService($this->leadRepo, $this->userRepo, $this->stageRepo);
    }

    public function testCreateWithEmptyBodyReturnsAllValidationErrors(): void {
        try { $this->service->create([]); }
        catch (ValidationException $e) {
            $this->assertArrayHasKey('userId', $e->getErrors());
            $this->assertArrayHasKey('name', $e->getErrors());
            $this->assertArrayHasKey('email', $e->getErrors());
            return;
        }
        $this->fail('ValidationException was not thrown');
    }

    public function testDeleteSetsDeletedAtAndCallsSave(): void {
        $lead = $this->existingLead();
        $leadRepo = $this->createMock(LeadRepository::class);
        $leadRepo->method('findById')->willReturn($lead);
        $leadRepo->expects($this->once())->method('save');
        (new LeadService($leadRepo, $this->userRepo, $this->stageRepo))->delete(1);
        $this->assertNotNull($lead->getDeletedAt());
    }
}

class LeadIntegrationTest extends IntegrationTestCase {

    public function testPostCreatesLeadAndReturns201(): void {
        $response = $this->json('POST', '/lead', $this->validLead());
        $this->assertSame(201, $this->statusCode());
        $this->assertSame('Joao Silva', $response['data']['name']);
    }

    public function testDeleteSoftDeletesLeadAndReturns204(): void {
        $lead = new Lead();
        $lead->setUser($this->user)->setStage($this->stage)
             ->setName('Joao')->setCompany('Empresa')
             ->setEmail('joao@email.com')->setPhone('85999999999')->setValue('1000.00');
        $this->em->persist($lead);
        $this->em->flush();

        $this->json('DELETE', "/lead/{$lead->getId()}");
        $this->assertSame(204, $this->statusCode());

        $this->em->clear();
        $found = $this->em->find(Lead::class, $lead->getId());
        $this->assertNotNull($found->getDeletedAt());
    }
}
